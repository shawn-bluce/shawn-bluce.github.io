    import time

    #get date 获取当前日期
    year  = time.strftime('%Y', time.localtime(time.time()))

    day   = time.strftime('%d', time.localtime(time.time()))
    day = (int)(day)
    week  = time.strftime('%w', time.localtime(time.time()))

    week = (int)(week)
    for i in range(0, day - 1):
